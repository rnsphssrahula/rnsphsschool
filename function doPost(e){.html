function doPost(e){ var sheet = SpreadsheetApp.getActiveSheet(); var data = JSON.parse(e.postData.contents); sheet.appendRow([ data.name, data.dob, data.gender, data.class, data.father, data.phone, data.email, data.address, data.city, data.district, data.pincode, new Date() ]); return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); }
